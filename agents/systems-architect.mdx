    ```c
    // Rate-Monotonic Scheduling (RMS)
    // Priority assigned based on period (shorter period = higher priority)
    
    typedef struct {
        int period;          // Task period (ms)
        int execution_time;  // Worst-case execution time (ms)
        int deadline;        // Relative deadline (ms)
        int priority;        // Priority (higher number = higher priority)
    } Task;
    
    // Schedulability test: U = Σ(Ci/Ti) ≤ n(2^(1/n) - 1)
    float calculate_utilization(Task tasks[], int n) {
        float total_util = 0.0;
        for (int i = 0; i < n; i++) {
            total_util += (float)tasks[i].execution_time / tasks[i].period;
        }
        return total_util;
    }
    
    bool is_schedulable_rms(Task tasks[], int n) {
        float utilization = calculate_utilization(tasks, n);
        float bound = n * (pow(2.0, 1.0/n) - 1);
        return utilization <= bound;
    }
    ```

### 3. Implement Deterministic Memory Management

Eliminate unpredictable latency:

```c
// Memory pool for deterministic allocation
typedef struct {
    void* pool;
    size_t block_size;
    size_t num_blocks;
    uint8_t* free_list;
} MemoryPool;

// Initialize memory pool
MemoryPool* create_pool(size_t block_size, size_t num_blocks) {
    MemoryPool* pool = malloc(sizeof(MemoryPool));
    pool->block_size = block_size;
    pool->num_blocks = num_blocks;
    
    // Pre-allocate all memory
    pool->pool = malloc(block_size * num_blocks);
    pool->free_list = malloc(num_blocks);
    
    // Initialize free list
    for (size_t i = 0; i < num_blocks; i++) {
        pool->free_list[i] = 1;  // Mark as free
    }
    
    return pool;
}

// O(1) deterministic allocation
void* pool_alloc(MemoryPool* pool) {
    for (size_t i = 0; i < pool->num_blocks; i++) {
        if (pool->free_list[i]) {
            pool->free_list[i] = 0;  // Mark as used
            return (char*)pool->pool + (i * pool->block_size);
        }
    }
    return NULL;  // Pool exhausted
}

// O(1) deterministic deallocation
void pool_free(MemoryPool* pool, void* ptr) {
    size_t index = ((char*)ptr - (char*)pool->pool) / pool->block_size;
    pool->free_list[index] = 1;  // Mark as free
}

// Lock-free stack for multi-threaded access
typedef struct Node {
    void* data;
    struct Node* next;
} Node;

typedef struct {
    _Atomic(Node*) head;
} LockFreeStack;

void push(LockFreeStack* stack, void* data) {
    Node* node = malloc(sizeof(Node));
    node->data = data;
    Node* old_head = atomic_load(&stack->head);
    do {
        node->next = old_head;
    } while (!atomic_compare_exchange_weak(&stack->head, &old_head, node));
}
```
