    ```python
    from functools import wraps
    from typing import TypeVar, Callable
    import inspect
    
    F = TypeVar('F', bound=Callable)
    
    def typed_cache(func: F) -> F:
        """Decorator that caches based on type hints."""
        cache = {}
        sig = inspect.signature(func)
        
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Validate types
            bound = sig.bind(*args, **kwargs)
            for name, value in bound.arguments.items():
                param = sig.parameters[name]
                if param.annotation != inspect.Parameter.empty:
                    if not isinstance(value, param.annotation):
                        raise TypeError(f"{name} must be {param.annotation}")
            
            # Cache based on args
            key = (args, tuple(sorted(kwargs.items())))
            if key not in cache:
                cache[key] = func(*args, **kwargs)
            return cache[key]
        
        return wrapper
    ```

    def bidirectional_generator() -> Generator[int, str, None]:

## Advanced Patterns

<CodeGroup>
```python Descriptor Protocol
class TypedProperty:
    """Type-validated property descriptor."""
    def __init__(self, name, expected_type):
        self.name = name
        self.expected_type = expected_type
    
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        return obj.__dict__.get(self.name)
    
    def __set__(self, obj, value):
        if not isinstance(value, self.expected_type):
            raise TypeError(f"{self.name} must be {self.expected_type}")
        obj.__dict__[self.name] = value

class Person:
    name = TypedProperty('name', str)
    age = TypedProperty('age', int)
```

```python Metaclass Pattern
class Singleton(type):
    """Singleton metaclass."""
    _instances = {}
    
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class Database(metaclass=Singleton):
    def __init__(self):
        self.connection = None
```

```python Async Context Manager
class AsyncDatabaseConnection:
    async def __aenter__(self):
        self.conn = await create_connection()
        return self.conn
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.conn.close()
        if exc_type is not None:
            await self.conn.rollback()

# Usage
async with AsyncDatabaseConnection() as conn:
    await conn.execute("SELECT * FROM users")
```
