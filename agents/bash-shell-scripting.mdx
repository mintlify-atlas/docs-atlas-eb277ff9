    #!/usr/bin/env bash
    
    # Usage function
    usage() {
        cat <<EOF
    Usage: $0 [OPTIONS]
    
    Options:
        -h, --help          Show this help message
        -v, --verbose       Enable verbose output
        -f, --file FILE     Input file path
        -n, --number NUM    Number of iterations
    EOF
        exit 0
    }
    
    # Parse arguments
    VERBOSE=0
    FILE=""
    NUMBER=10
    
    while [[ $# -gt 0 ]]; do
        case $1 in
            -h|--help)
                usage
                ;;
            -v|--verbose)
                VERBOSE=1
                shift
                ;;
            -f|--file)
                FILE="$2"
                shift 2
                ;;
            -n|--number)
                NUMBER="$2"
                shift 2
                ;;
            *)
                echo "Unknown option: $1" >&2
                usage
                ;;
        esac
    done
    
    # Validate required arguments
    if [[ -z "$FILE" ]]; then
        echo "Error: --file is required" >&2
        exit 1
    fi

    ```bash
    #!/usr/bin/env bash
    
    # Logging setup
    readonly LOG_FILE="${LOG_FILE:-/var/log/myapp.log}"
    readonly LOG_LEVEL="${LOG_LEVEL:-INFO}"
    
    # Log levels
    declare -A LOG_LEVELS=([DEBUG]=0 [INFO]=1 [WARN]=2 [ERROR]=3)
    
    log() {
        local level=$1
        shift
        local message="$*"
        local timestamp=$(date '+%Y-%m-%d %H:%M:%S')
        
        # Check log level
        if [[ ${LOG_LEVELS[$level]} -ge ${LOG_LEVELS[$LOG_LEVEL]} ]]; then
            echo "[$timestamp] [$level] $message" | tee -a "$LOG_FILE" >&2
        fi
    }
    
    log_debug() { log DEBUG "$@"; }
    log_info()  { log INFO "$@"; }
    log_warn()  { log WARN "$@"; }
    log_error() { log ERROR "$@"; }
    ```

    #!/usr/bin/env bash
    
    # Process files in parallel
    process_file() {
        local file=$1
        echo "Processing $file..."
        # Processing logic here
    }
    export -f process_file
    
    # Using GNU parallel
    find . -name "*.txt" | parallel -j 4 process_file
    
    # Alternative: background jobs with wait
    max_jobs=4
    job_count=0
    
    for file in *.txt; do
        process_file "$file" &
        ((job_count++))
        
        # Wait when max jobs reached
        if [[ $job_count -ge $max_jobs ]]; then
            wait -n  # Wait for any job to finish
            ((job_count--))
        fi
    done
    
    # Wait for remaining jobs
    wait

## Production Deployment Script Example

```bash
#!/usr/bin/env bash
set -euo pipefail

# Configuration
readonly APP_NAME="myapp"
readonly DEPLOY_DIR="/var/www/${APP_NAME}"
readonly RELEASES_DIR="${DEPLOY_DIR}/releases"
readonly CURRENT_LINK="${DEPLOY_DIR}/current"
readonly KEEP_RELEASES=5

# Health check
health_check() {
    local url=$1
    local max_attempts=30
    local attempt=0
    
    while [[ $attempt -lt $max_attempts ]]; do
        if curl -sf "$url" >/dev/null; then
            return 0
        fi
        ((attempt++))
        sleep 2
    done
    return 1
}

# Deploy new release
deploy() {
    local version=$1
    local release_dir="${RELEASES_DIR}/${version}"
    
    echo "Deploying version $version..."
    
    # Extract release
    mkdir -p "$release_dir"
    tar -xzf "release-${version}.tar.gz" -C "$release_dir"
    
    # Atomic symlink switch
    ln -sfn "$release_dir" "${CURRENT_LINK}.tmp"
    mv -T "${CURRENT_LINK}.tmp" "$CURRENT_LINK"
    
    # Health check
    if ! health_check "http://localhost:8080/health"; then
        echo "Health check failed, rolling back..."
        rollback
        return 1
    fi
    
    # Cleanup old releases
    cleanup_old_releases
    
    echo "Deployment successful"
}

# Rollback to previous release
rollback() {
    local previous=$(ls -t "$RELEASES_DIR" | sed -n 2p)
    if [[ -z "$previous" ]]; then
        echo "No previous release found" >&2
        return 1
    fi
    
    echo "Rolling back to $previous..."
    ln -sfn "${RELEASES_DIR}/${previous}" "$CURRENT_LINK"
}

# Cleanup old releases
cleanup_old_releases() {
    cd "$RELEASES_DIR"
    ls -t | tail -n +$((KEEP_RELEASES + 1)) | xargs rm -rf
}

deploy "$1"
```
