<Tabs>
  <Tab title="Worker Pool">
    ```go
    type WorkerPool struct {
        workers   int
        jobs      chan Job
        results   chan Result
        ctx       context.Context
        cancel    context.CancelFunc
        wg        sync.WaitGroup
    }
    
    func NewWorkerPool(workers int) *WorkerPool {
        ctx, cancel := context.WithCancel(context.Background())
        return &WorkerPool{
            workers: workers,
            jobs:    make(chan Job, workers*2),
            results: make(chan Result, workers*2),
            ctx:     ctx,
            cancel:  cancel,
        }
    }
    
    func (p *WorkerPool) Start() {
        for i := 0; i < p.workers; i++ {
            p.wg.Add(1)
            go p.worker()
        }
    }
    
    func (p *WorkerPool) worker() {
        defer p.wg.Done()
        for {
            select {
            case job := <-p.jobs:
                result := job.Process()
                p.results <- result
            case <-p.ctx.Done():
                return
            }
        }
    }
    ```
  </Tab>
  
  <Tab title="Fan-In/Fan-Out">
    ```go
    // Fan-out: distribute work to multiple goroutines
    func fanOut(input <-chan int, workers int) []<-chan int {
        channels := make([]<-chan int, workers)
        for i := 0; i < workers; i++ {
            channels[i] = process(input)
        }
        return channels
    }
    
    // Fan-in: merge multiple channels into one
    func fanIn(channels ...<-chan int) <-chan int {
        out := make(chan int)
        var wg sync.WaitGroup
        
        for _, ch := range channels {
            wg.Add(1)
            go func(c <-chan int) {
                defer wg.Done()
                for val := range c {
                    out <- val
                }
            }(ch)
        }
        
        go func() {
            wg.Wait()
            close(out)
        }()
        
        return out
    }
    ```
  </Tab>

    // Stage 1: Generate numbers
    func generate(nums ...int) <-chan int {
        out := make(chan int)
        go func() {
            defer close(out)
            for _, n := range nums {
                out <- n
            }
        }()
        return out
    }
    
    // Stage 2: Square numbers
    func square(in <-chan int) <-chan int {
        out := make(chan int)
        go func() {
            defer close(out)
            for n := range in {
                out <- n * n
            }
        }()
        return out
    }

## Best Practices

<CardGroup cols={2}>
  <Card title="Goroutine Management" icon="arrows-spin">
    - Always ensure goroutines can exit
    - Use context for cancellation
    - Avoid goroutine leaks with proper cleanup
    - Monitor goroutine count in production
  </Card>
  <Card title="Channel Usage" icon="arrows-left-right">
    - Close channels from sender side
    - Use select with default for non-blocking ops
    - Avoid sending to closed channels
    - Prefer buffered channels for known capacity
  </Card>
  <Card title="Synchronization" icon="lock">
    - Prefer channels over shared memory
    - Use sync.Once for initialization
    - Apply RWMutex for read-heavy workloads
    - Consider atomic operations for simple cases
  </Card>
  <Card title="Error Handling" icon="triangle-exclamation">
    - Return errors, don't panic
    - Use errgroup for goroutine error handling
    - Implement proper error wrapping
    - Log errors with context
  </Card>
</CardGroup>

<Warning>
**Common Pitfall**: Goroutine leaks occur when goroutines block forever waiting on channels. Always ensure cleanup paths exist using context cancellation.
</Warning>
