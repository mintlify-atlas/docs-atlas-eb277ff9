# Validation System

ccprompts uses a sophisticated multi-dimensional validation engine (`scripts/validate-commands.js`) that validates commands across four key dimensions: structure, security, quality, and performance.

## Overview

The validation system ensures that all 70+ commands meet rigorous standards before deployment. Each command receives a numeric score (0-100) reflecting its validation pass rate across all dimensions.

<Note>
**Current Performance**: Discovery in &lt;100ms, total validation in &lt;2s (currently achieving 3ms + 28ms)
</Note>

## Dimension 1: Structural Validation

Structural validation ensures commands follow the correct markdown format and include all required sections.

### Required Sections

```javascript
validateCommandStructure(content, filename) {
  const metadata = this.extractCommandMetadata(content, filename);

  const hasDescription = content.includes('## Description') ||
                        content.includes('# ') ||
                        content.match(/^description:/m);

  if (!hasDescription) {
    this.errors.push(`${filename}: Command file missing description`);
  }

  const hasUsageSection = content.includes('## Usage');
  const hasExamplesSection = content.includes('## Examples');
  
  if (!hasUsageSection && !hasExamplesSection) {
    this.warnings.push(`${filename}: Consider adding sections: ## Usage, ## Examples`);
  }
}
```

### XML Structure Validation

The system validates XML configuration blocks for proper tag balance:

```javascript
validateXMLStructure(content, filename) {
  const requiredSections = ['<role>', '<activation>', '<instructions>'];
  const missingSections = [];

  requiredSections.forEach(section => {
    if (!content.includes(section)) {
      missingSections.push(section);
    }
  });

  if (missingSections.length > 0) {
    this.errors.push(`${filename}: Missing XML sections: ${missingSections.join(', ')}`);
    return false;
  }

  // Validate tag balance
  const tagStack = [];
  const xmlTagRegex = /<\/?([a-zA-Z][a-zA-Z0-9_-]*)(?:\s[^>]*)?>|<!--[\s\S]*?-->/g;
  let match;

  while ((match = xmlTagRegex.exec(contentWithoutCodeBlocks)) !== null) {
    const fullTag = match[0];
    const tagName = match[1];

    if (fullTag.startsWith('<!--') || fullTag.endsWith('/>')) {
      continue;
    }

    if (fullTag.startsWith('</')) {
      const expectedTag = tagStack.pop();
      if (expectedTag !== tagName) {
        this.errors.push(
          `${filename}: Mismatched XML tags - expected </${expectedTag}>, found </${tagName}>`
        );
        return false;
      }
    } else {
      tagStack.push(tagName);
    }
  }

  if (tagStack.length > 0) {
    this.errors.push(`${filename}: Unclosed XML tags: ${tagStack.join(', ')}`);
    return false;
  }

  return true;
}
```

### Metadata Extraction

The validator extracts and validates command metadata:

```javascript
extractCommandMetadata(content, filename) {
  const relativePath = path.relative(process.cwd(), filename);
  const commandName = path.basename(filename, '.md');

  // Extract title from H1 heading
  const titleMatch = content.match(/^#\s+(.+)$/m);
  const title = titleMatch ? titleMatch[1].trim() : commandName;

  // Extract description from Description section
  let description = this.extractMarkdownSection(content, '## Description');
  
  // Determine category and phase from path
  const category = this.extractCategoryFromPath(relativePath);
  const phase = this.extractPhaseFromPath(relativePath);

  // Extract usage, parameters, examples
  const usage = this.extractUsage(content);
  const parameters = this.extractParameters(content);
  const examples = this.extractExamples(content);
  const safetyLevel = this.determineSafetyLevel(content);

  return {
    id: commandName,
    name: title,
    category,
    phase,
    description,
    usage,
    parameters,
    examples,
    safety_level: safetyLevel,
    file_path: relativePath,
    last_modified: new Date().toISOString()
  };
}
```

## Dimension 2: Security Validation

Security validation detects hardcoded secrets, dangerous patterns, and potential vulnerabilities.

### Secret Detection

```javascript
validateSecurity(content, filename) {
  const codeBlocks = this.extractCodeBlocks(content);
  const combinedCode = codeBlocks.join('\n');

  const securityPatterns = [
    {
      pattern: /password\s*=\s*["'][^"']{8,}["']/gi,
      message: 'Hardcoded password detected',
      skipIfIncludes: ['example', 'placeholder', 'your-', 'REPLACE_WITH']
    },
    {
      pattern: /api[_-]?key\s*=\s*["'][^"']{16,}["']/gi,
      message: 'Hardcoded API key detected',
      skipIfIncludes: ['example', 'placeholder', 'your-', 'REPLACE_WITH']
    },
    {
      pattern: /secret\s*=\s*["'][^"']{8,}["']/gi,
      message: 'Hardcoded secret detected',
      skipIfIncludes: ['example', 'placeholder', 'your-', 'REPLACE_WITH']
    },
    {
      pattern: /token\s*=\s*["'][^"']{16,}["']/gi,
      message: 'Hardcoded token detected',
      skipIfIncludes: ['example', 'placeholder', 'your-', 'REPLACE_WITH']
    }
  ];

  securityPatterns.forEach(({ pattern, message, skipIfIncludes }) => {
    const matches = combinedCode.match(pattern) || [];

    matches.forEach(match => {
      if (skipIfIncludes && skipIfIncludes.some(skip => 
        match.toLowerCase().includes(skip.toLowerCase())
      )) {
        return;
      }

      this.errors.push(`${filename}: SECURITY - ${message}`);
      this.stats.securityIssues++;
    });
  });
}
```

### Dangerous Pattern Detection

Integrates with the safety pattern system:

```javascript
analyzeDangerousPatterns(content, filename) {
  const findings = [];
  const codeBlocks = this.extractCodeBlocks(content);

  codeBlocks.forEach((block, index) => {
    const allPatterns = safetyPatterns.getAllPatterns();

    for (const { pattern, severity, message, skipIfIncludes } of allPatterns) {
      const regex = new RegExp(pattern.source, pattern.flags);
      const matches = blockContent.match(regex) || [];

      if (matches.length > 0) {
        findings.push({
          filename,
          blockIndex: index,
          pattern: pattern.source,
          severity,
          message,
          matches: matches.slice(0, 3),
          codeSnippet: blockContent.substring(0, 100)
        });
      }
    }
  });

  return findings;
}
```

## Dimension 3: Quality Validation

Quality validation assesses educational value, clarity, and completeness.

### Quality Scoring

```javascript
validateCommandQuality(content, filename, commandType = null) {
  const detectedType = commandType || this.determineCommandType(filename, content);
  let qualityScore = 100;

  const qualityChecks = {
    command: [
      {
        name: 'Has usage section',
        test: c => c.includes('## Usage'),
        weight: 25,
        message: 'Missing Usage section'
      },
      {
        name: 'Has description section',
        test: c => c.includes('## Description'),
        weight: 20,
        message: 'Missing Description section'
      },
      {
        name: 'Has parameters section',
        test: c => c.includes('## Parameters'),
        weight: 20,
        message: 'Missing Parameters section'
      },
      {
        name: 'Has examples section',
        test: c => c.includes('## Examples'),
        weight: 25,
        message: 'Missing Examples section'
      },
      {
        name: 'Professional formatting',
        test: c => !/(TODO|FIXME|XXX|HACK)/i.test(c),
        weight: 10,
        message: 'Contains TODO/FIXME markers'
      }
    ]
  };

  const checks = qualityChecks[detectedType] || qualityChecks['default'];

  checks.forEach(check => {
    if (!check.test(content)) {
      qualityScore -= check.weight;
      this.warnings.push(
        `${filename} [${detectedType}]: ${check.message} (-${check.weight} points)`
      );
    }
  });

  this.stats.qualityScores.push(qualityScore);
  return qualityScore;
}
```

## Dimension 4: Performance Validation

Performance validation ensures the validation system itself is fast and efficient.

### Performance Metrics

```javascript
const performanceMetrics = {
  discovery_time: 0,           // File discovery time
  validation_time: 0,          // Total validation time
  registry_generation_time: 0, // Command registry generation
  file_processing_times: [],   // Per-file timing
  memory_usage: process.memoryUsage()
};

// Discovery phase
const discoveryStart = Date.now();
const markdownFiles = this.findMarkdownFiles(projectRoot);
performanceMetrics.discovery_time = Date.now() - discoveryStart;

// Validation phase
const validationStart = Date.now();
for (const file of markdownFiles) {
  const fileStart = Date.now();
  await this.validateFile(file);
  performanceMetrics.file_processing_times.push({
    file: file,
    duration: Date.now() - fileStart
  });
}
performanceMetrics.validation_time = Date.now() - validationStart;
```

## System Integrity Validation

Beyond individual commands, the system validates overall project integrity:

```javascript
validateSystemIntegrity() {
  // Verify package.json has validate script
  const packageJson = JSON.parse(fs.readFileSync('package.json', 'utf8'));
  if (!packageJson.scripts || !packageJson.scripts.validate) {
    this.warnings.push('package.json: Missing validate script');
  }

  // Check required files exist
  const requiredFiles = ['.gitignore', 'README.md', 'CONTRIBUTING.md'];
  requiredFiles.forEach(file => {
    if (!fs.existsSync(file)) {
      this.warnings.push(`Missing required file: ${file}`);
    }
  });

  // Validate command count
  const commandDir = path.join(process.cwd(), '.claude', 'commands');
  const expectedCommandCount = process.env.EXPECTED_COMMAND_COUNT
    ? parseInt(process.env.EXPECTED_COMMAND_COUNT, 10)
    : null;

  if (expectedCommandCount !== null) {
    const commandFiles = this.countMarkdownFiles(commandDir);
    if (commandFiles !== expectedCommandCount) {
      this.errors.push(
        `Expected ${expectedCommandCount} commands, found ${commandFiles}`
      );
    }
  }
}
```

## Quality Gates for Commits

Before committing changes, ensure all validations pass:

<Steps>
  <Step title="Run Full Validation">
    ```bash
    bun run validate
    ```
    Should find exactly 70 commands with no structural errors.
  </Step>
  
  <Step title="Check Markdown Linting">
    ```bash
    bun run lint
    ```
    Should pass with no style violations.
  </Step>
  
  <Step title="Validate Links">
    ```bash
    bun run check-links
    ```
    All markdown links must be valid.
  </Step>
  
  <Step title="Run Tests (if modified)">
    ```bash
    bun run test
    ```
    Jest tests + validation must pass.
  </Step>
  
  <Step title="Security Scan">
    ```bash
    bun run security-scan
    ```
    Check for new security warnings.
  </Step>
</Steps>

<Warning>
If adding or removing commands, update phase counts in `.claude-plugin/plugin.json` to match the actual command count.
</Warning>
